在宝塔：
点击添加站点
域名填写ip地址，不用写端口
目录设置为/www/wwwroot/api_backend
PHP版本纯静态

（这部分内容可以在后面手动修改 Nginx 配置文件）
代理名称：api_backend
目标 URL：http://127.0.0.1:8000
发送域名：$host
代理目录：/api/ (手动修改配置文件，或者在高级设置里)

手动修改：
打开网站设置 -> 配置文件，找到 server 块，修改成这样：
server {
    listen 80;
    server_name 120.26.177.237;

    # 1. 前端静态文件配置
    # 定义网站根目录为前端 dist 文件夹
    root /www/wwwroot/club-frontend/dist;
    index index.html index.htm;

    # 处理前端路由 (History 模式)
    location / {
        try_files $uri $uri/ /index.html;
    }

    # 2. 后端 API 反向代理
    location /api/ {
        proxy_pass http://127.0.0.1:8000/api/;  # 转发给本地后端
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header REMOTE-HOST $remote_addr;

        # 增加超时时间
        proxy_connect_timeout 300s;
        proxy_send_timeout 300s;
        proxy_read_timeout 300s;
    }

    # 3. 宝塔默认的安全配置 (保留)
    #CERT-APPLY-CHECK--START
    include /www/server/panel/vhost/nginx/well-known/120.26.177.237.conf;
    #CERT-APPLY-CHECK--END

    #SSL-START
    #error_page 404/404.html;
    #SSL-END

    error_page 404 /404.html;

    # 日志配置
    access_log /www/wwwlogs/120.26.177.237.log;
    error_log  /www/wwwlogs/120.26.177.237.error.log;

    # 禁止访问敏感文件 (宝塔默认规则，建议保留)
    location ~* "(
        \.user.ini|\.htaccess|\.htpasswd|\.env.*|\.project|\.bashrc|
        \.bash_profile|\.bash_logout|\.DS_Store|\.gitignore|
        \.gitattributes|LICENSE|README\.md|CLAUDE\.md|CHANGELOG\.md|
        CONTRIBUTING\.md|TODO\.md|FAQ\.md|composer\.json|composer\.lock|
        package(-lock)?\.json|yarn\.lock|pnpm-lock\.yaml|\.\w+~|
        \.swp|\.swo|\.bak(up)?|\.old|\.tmp|\.temp|\.log|\.sql(\.gz)?|
        docker-compose\.yml|docker\.env|Dockerfile|\.csproj|\.sln|
        Cargo\.toml|Cargo\.lock|go\.mod|go\.sum|phpunit\.xml|
        pom\.xml|build\.gradl|pyproject\.toml|requirements\.txt|
        application(-\w+)?\.(ya?ml|properties)
    )$" {
        return 404;
    }

    location ~* "/(
        \.git|\.svn|\.bzr|\.vscode|\.claude|\.idea|\.ssh|\.github|
        \.npm|\.yarn|\.pnpm|\.cache|\.husky|\.turbo|\.next|\.nuxt|
        node_modules|runtime
    )/" {
        return 404;
    }

    #去掉了 include enable-php-00.conf;
    # 因为这是纯前端 + API 项目，不需要解析 PHP
}
保存即可
