## 📁 新增文件

### system.ts - 系统配置 API 模块

**文件**: `frontend/src/api/modules/system.ts`

**内容**:
```typescript
import { get } from '../request'

// 系统角色
export interface SystemRole {
  code: string
  name: string
  description: string
  icon?: string
  sort_order: number
}

// 社团分类
export interface ClubCategory {
  id: string
  name: string
  description?: string
  sort_order: number
}

// 系统配置
export interface SystemConfig {
  key: string
  value: any
  description?: string
  type: 'string' | 'number' | 'boolean' | 'array' | 'object'
}

// 获取系统角色列表
export function getSystemRoles() {
  return get<SystemRole[]>('/api/system/roles')
}

// 获取社团分类列表
export function getClubCategories() {
  return get<ClubCategory[]>('/api/system/club-categories')
}

// 获取系统配置
export function getSystemConfig(key?: string) {
  return get<SystemConfig | SystemConfig[]>('/api/system/config', { key })
}

// 获取多个系统配置
export function getSystemConfigs(keys: string[]) {
  return get<Record<string, any>>('/api/system/configs', { keys: keys.join(',') })
}

// 获取学校列表
export function getSchools() {
  return get<Array<{ code: string; name: string }>>('/api/system/schools')
}
```

from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from app.api.deps import get_current_active_user, get_db
from app.models.user import User

router = APIRouter()

@router.get("/roles")
async def get_system_roles(
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_active_user)
):
    """获取系统角色列表"""
    # 方案1: 从数据库读取
    roles = db.query(Role).order_by(Role.sort_order).all()
    return [
        {
            "code": role.code,
            "name": role.name,
            "description": role.description,
            "icon": role.icon,
            "sort_order": role.sort_order
        }
        for role in roles
    ]

    # 方案2: 硬编码（临时方案）
    # return [

    #     {"code": "admin", "name": "社团管理员", "description": "管理社团招新、面试安排等", "sort_order": 3},

@router.get("/club-categories")

@router.get("/schools")
async def get_schools(

    schools = db.query(School).all()
